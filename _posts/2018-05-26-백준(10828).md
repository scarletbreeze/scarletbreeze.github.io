## 백준 스택 구현 문제 - 10828

### 배열로 풀기
```
 #include<stdio.h>
#include<string.h>
int main(void){
    int N = 0;
    
    unsigned int size = 0;
    int stack[10000] = {0};
    char cmd[6] = {0};
    int item = 0;
    
    scanf("%d\n",&N);
    for(int i =0; i<N; i++){
        scanf("%s",cmd);
        //pop
        if(strcmp(cmd, "pop") == 0){
            if(size == 0){
                printf("-1\n");
                continue;
            }
            printf("%d\n",stack[size-1]);
            size--;
        }
        
        //push
        else if(strcmp(cmd, "push") == 0){
            scanf("%d\n", &item);
            stack[size] = item;
            size++;
        }
        
        //top
        else if(strcmp(cmd, "top") == 0){
            if(size == 0){
                printf("-1\n");
                continue;
            }
            printf("%d\n",stack[size-1]);
        }
        
        //empty
        else if(strcmp(cmd, "empty") == 0){
            printf("%d\n", size == 0); // 스택이 비어있으면 1, 아니면 0을 출력한다.
        }
        else{
            printf("%d\n",size);
        }
    }
    return 0;
}


```
