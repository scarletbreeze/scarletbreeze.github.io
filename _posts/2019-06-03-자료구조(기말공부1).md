## 복습할 내용 체크

- degree가 뭐였지 ? 자식들의 수
- arrayList?
  1. 길이가 고정된 형태로 선언될 필요 없어
  2. 정렬도 매우 쉽다.
     `import java.util.Collections` -> `Collections.sort(정렬할 리스트)`;
- tree의 size? -> 트리 내의 노드의 개수. -> 변수 선언해서 해줘야함
- external node = 외부노드 => 자식이 없어
- addRoot -> 기존 루트를 새로 생성한다.
- addNode -> root의 child로 넣는다.
- addNode 할 때, root의 child로 Object를 넣냐 아니면 MyNode를 넣냐 ? -> MyNode를 넣는다.
- arrayList -> set ? set은 그 인덱스에 해당하는 노드를 교체해준다 (replace)
- 노드를 추가할 때, tree를 기준으로 추가해야한다. ex) tree.addchild()

### MyNode.java

```java
package tree;

import java.util.ArrayList;

public class MyNode {
	private Object element;
	private MyNode parent;
	private ArrayList<Object> children;

	// implement constructors

	public MyNode(){
		element = null;
		parent = null;
		children = null;
	}
	public MyNode(Object e) {
		element = e;
		parent = null;
		children = null;
	}
	public MyNode(Object e, MyNode p, ArrayList<Object> c) {
		this.element= e;
		this.parent = p;
		this.children = c;
	}

	//implement the following methods

	public Object element() {
		return this.element;
	}

	public MyNode parent() {
		return this.parent;
	}
	public ArrayList<Object> children(){
		return this.children;
	}
	public int degree() {
		return this.children.size();
	}
	public void setElement(Object e) {
		this.element = e;
	}
	public void setParent(MyNode p) {
		this.parent = p;
	}
	public void setChildren(ArrayList<Object> c) {
		this.children = c;
	}
}
```

### MyTree.java

```java
package tree;

import java.util.ArrayList;

public class MyTree {
	private MyNode root;
	private int TreeSize;

	//implement constructors

	public MyTree() {
		this.root = null;
		this.TreeSize = 0;
	}

	public MyTree(Object e) {
		this.root = new MyNode(e);
		this.root.setChildren(new ArrayList<Object>());
		this.TreeSize = 1;
	}

	//
	public int size() {
		return this.TreeSize;
	}

	public MyNode root() {
		return this.root;
	}
	public ArrayList<Object> children(MyNode v) {
		return v.children();
	}
	public boolean isExternal(MyNode v) {
		return v.children().size() == 0;
	}

	// addRoot 기존의 루트 바꾼다 보다는 새로운 루트 생성. 새로 생긴 루트 반환
	public MyNode addRoot(Object e) {
		this.root = new MyNode(e);
		this.root.setChildren(new ArrayList<Object>());
		this.TreeSize = 1;
		return this.root;
	}
	// root의 child로 넣는다. 새로 생성한 노드 반환
	public MyNode addNode(Object e) {
		MyNode childNode = new MyNode(e,root(),new ArrayList<Object>());
		this.root.children().add(childNode);
		this.TreeSize++;
		return childNode;
	}

	// v의 child로 addChild
	public MyNode addChild(MyNode v, Object e) {
		MyNode childNode = new MyNode(e,v, new ArrayList<Object>());
		v.children().add(childNode);
		this.TreeSize++;
		return childNode;
	}

	//v의 몇 번째 child로 add
	public MyNode addChild(MyNode v, int i, Object e) {
		MyNode childNode = new MyNode(e,v,new ArrayList<Object>());
		v.children().add(i, e);
		this.TreeSize++;
		return childNode;
	}

	public MyNode setChild(MyNode v, int i, Object e) {
		MyNode childNode = new MyNode(e,v,new ArrayList<Object>());
		v.children().set(i, e);
		return childNode;
	}
	public MyNode removeChild(MyNode v, int i) {
		this.TreeSize--;
		return (MyNode) v.children().remove(i);
	}
}


```

### Main.java

```java
package tree;

public class Main {

	public static void main(String[] args) {
		MyTree Tree = new MyTree("Computers'R Us");
		System.out.println("[Level 0]");
		System.out.println(Tree.root().element());
		System.out.println();

		MyNode Sales = Tree.addNode("Sales");
		MyNode Manu = Tree.addNode("Manufacturing");
		MyNode RandD = Tree.addNode("R&D");

		System.out.println("[Level 1]");
		System.out.println(Sales.element() + "," + Manu.element() + "," + RandD.element());
		System.out.println();

		MyNode Us = Tree.addChild(Sales, "US");
		MyNode Inte = Tree.addChild(Sales, "International");

		MyNode Laptops = Tree.addChild(Manu, "Laptops");
		MyNode Desktops = Tree.addChild(Manu, "Desktops");

		System.out.println("[Level 2]");
		System.out.println(Us.element() + "," + Inte.element() + "," + Laptops.element() + "," + Desktops.element());
		System.out.println();

		MyNode Europe = Tree.addChild(Inte, "Europe");
		MyNode Asia = Tree.addChild(Inte, "Asia");
		MyNode Canada = Tree.addChild(Inte, "Canada");

		System.out.println("[Level 3]");
		System.out.println(Europe.element() + "," + Inte.element() + "," + Asia.element() + "," + Canada.element());
		System.out.println();

		System.out.println("Tree Size = Total " + Tree.size() +"Nodes");


	}
}
```
